Practical-1
(BLINK OF SINGLE LED)
import RPi.GPIO as GPIO
import time
GPIO.setmode(GPIO.BOARD)
GPIO.setup(7,GPIO.OUT)
for i in range(10):
         GPIO.output(7,True)
         print("LED IS FINALLY ON")

         time.sleep(1)

         GPIO.output(7,False)
         print("LED IS OFF")

         time.sleep(1)
print("PROGRAM COMPLETE!")
GPIO.cleanup()

PREVIEW-IMAGE:

import time

preview_config = picam2.create_preview_configuration()
picam2.configure(preview_config)
picam2.start_preview(Preview.QTGL)

picam2.start()

VIDEO:
from picamera2 import Picamera2
import time

picam2 = Picamera2()

video_config = picam2.create_video_configuration()
picam2.configure(video_config)

picam2.start()
picam2.start_recording("video.h264")
time.sleep(10)
picam2.stop_recording()
picam2.stop()
print("video saved")














 
PRACTICAL-3
(4 LED PATTERN)
import RPi.GPIO as GPIO
import time

GPIO.setmode(GPIO.BOARD)
GPIO.setup(3, GPIO.OUT)
GPIO.setup(5, GPIO.OUT)
GPIO.setup(7, GPIO.OUT)
GPIO.setup(8, GPIO.OUT)

while True:
    GPIO.output(3, 1)
    GPIO.output(5, 1)
    GPIO.output(7, 1)
    GPIO.output(8, 0)
    time.sleep(2)
    GPIO.output(3, 0)
    GPIO.output(5, 0)
    GPIO.output(7, 1)
    GPIO.output(8, 1)
    time.sleep(1)
